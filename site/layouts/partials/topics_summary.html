{{- $topicsList := slice -}} {{- range $topic := . -}} {{- if not $topic.subtopics -}} {{- $topicsList = $topicsList | append (title $topic.topic) -}} {{- else -}} {{- range $subtopic := $topic.subtopics -}} {{- $topicsList = $topicsList | append (title $subtopic.subtopic) -}} {{- end -}} {{- end -}} {{- end -}} {{- delimit (first 3 (uniq $topicsList)) ", " -}}
